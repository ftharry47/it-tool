// ==================== CRITICAL FIX: GET SPREADSHEET FUNCTION ====================
// This function works across ALL browsers (Chrome, Safari, Firefox, Edge)
// It tries multiple methods to access the spreadsheet

function getSpreadsheet() {
  var ss = null;
  
  // Method 1: Use explicit Spreadsheet ID (MOST RELIABLE for web apps)
  if (SPREADSHEET_ID && SPREADSHEET_ID.trim() !== "") {
    try {
      ss = SpreadsheetApp.openById(SPREADSHEET_ID);
      if (ss) {
        return ss;
      }
    } catch (e) {
      Logger.log("Method 1 (openById) failed: " + e.toString());
    }
  }
  
  // Method 2: Try getActiveSpreadsheet (works when script is bound to sheet)
  try {
    ss = SpreadsheetApp.getActiveSpreadsheet();
    if (ss) {
      // Store the ID for future use
      var props = PropertiesService.getScriptProperties();
      props.setProperty('SPREADSHEET_ID', ss.getId());
      return ss;
    }
  } catch (e) {
    Logger.log("Method 2 (getActiveSpreadsheet) failed: " + e.toString());
  }
  
  // Method 3: Try to get from Script Properties (fallback)
  try {
    var props = PropertiesService.getScriptProperties();
    var storedId = props.getProperty('SPREADSHEET_ID');
    if (storedId) {
      ss = SpreadsheetApp.openById(storedId);
      if (ss) {
        return ss;
      }
    }
  } catch (e) {
    Logger.log("Method 3 (Script Properties) failed: " + e.toString());
  }
  
  // Method 4: Search by name in user's Drive (last resort)
  try {
    var files = DriveApp.getFilesByName("IT Support Tickets");
    if (files.hasNext()) {
      var file = files.next();
      ss = SpreadsheetApp.openById(file.getId());
      if (ss) {
        // Store the ID for future use
        var props = PropertiesService.getScriptProperties();
        props.setProperty('SPREADSHEET_ID', ss.getId());
        return ss;
      }
    }
  } catch (e) {
    Logger.log("Method 4 (Drive search) failed: " + e.toString());
  }
  
  // If all methods fail, throw a helpful error
  throw new Error("SPREADSHEET NOT FOUND: Please set SPREADSHEET_ID at the top of Code.gs. Open your Google Sheet, copy the ID from the URL (the long string between /d/ and /edit), and paste it in the SPREADSHEET_ID variable.");
}

function createSheetWithHeaders(ss, sheetName, headers, widths) {
  try {
    var sheet = ss.getSheetByName(sheetName);
    
    if (!sheet) {
      sheet = ss.insertSheet(sheetName);
      Logger.log("Created sheet: " + sheetName);
    } else {
      Logger.log("Sheet exists: " + sheetName);
    }
    
    // Set headers
    var headerRange = sheet.getRange(1, 1, 1, headers.length);
    headerRange.setValues([headers]);
    headerRange.setBackground("#1e40af");
    headerRange.setFontColor("#ffffff");
    headerRange.setFontWeight("bold");
    headerRange.setHorizontalAlignment("center");
    sheet.setFrozenRows(1);
    
    // Set column widths
    if (widths && widths.length) {
      widths.forEach(function(w, i) {
        if (i < headers.length) sheet.setColumnWidth(i + 1, w);
      });
    }
    
    return { sheet: sheetName, status: "ready", headers: headers.length };
    
  } catch (e) {
    Logger.log("Error creating sheet " + sheetName + ": " + e.toString());
    return { sheet: sheetName, status: "error", error: e.toString() };
  }
}

function updateAdminEmail(emailType, newEmail, updatedBy) {
  try {
    if (!newEmail || !isValidEmail(newEmail)) {
      return { success: false, error: 'Invalid email address' };
    }
    
    var settingName = emailType.toUpperCase() + '_EMAIL';
    if (emailType.toUpperCase() === 'SECONDARY' || emailType.toUpperCase() === 'TERTIARY') {
      settingName = emailType.toUpperCase() + '_ADMIN_EMAIL';
    }
    
    var result = setSetting(settingName, newEmail, updatedBy);
    
    if (result.success) {
      Logger.log("Admin email updated: " + emailType + " = " + newEmail);
      return { success: true, message: 'Email updated successfully' };
    }
    
    return result;
  } catch (e) {
    Logger.log("Error updating admin email: " + e.toString());
    return { success: false, error: e.toString() };
  }
}

function isValidEmail(email) {
  if (!email) return false;
  var regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return regex.test(email.toString().trim());
}

// ==================== SETTINGS MANAGEMENT ====================
function getSetting(settingName) {
  try {
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName(SETTINGS_SHEET);
    
    if (!sheet || sheet.getLastRow() < 2) {
      return null;
    }
    
    var data = sheet.getRange(2, 1, sheet.getLastRow() - 1, 2).getValues();
    
    for (var i = 0; i < data.length; i++) {
      if (data[i][0] === settingName) {
        return data[i][1];
      }
    }
    
    return null;
  } catch (e) {
    Logger.log("Error getting setting: " + e.toString());
    return null;
  }
}

function setSetting(settingName, value, updatedBy) {
  try {
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName(SETTINGS_SHEET);
    
    if (!sheet) {
      sheet = ss.insertSheet(SETTINGS_SHEET);
      sheet.getRange(1, 1, 1, SETTINGS_HEADERS.length).setValues([SETTINGS_HEADERS]);
      sheet.getRange(1, 1, 1, SETTINGS_HEADERS.length).setBackground("#1e40af").setFontColor("#ffffff").setFontWeight("bold");
    }
    
    var now = new Date();
    var lastRow = sheet.getLastRow();
    
    if (lastRow >= 2) {
      var data = sheet.getRange(2, 1, lastRow - 1, 1).getValues();
      
      for (var i = 0; i < data.length; i++) {
        if (data[i][0] === settingName) {
          sheet.getRange(i + 2, 2).setValue(value);
          sheet.getRange(i + 2, 3).setValue(now);
          sheet.getRange(i + 2, 4).setValue(updatedBy || 'System');
          return { success: true, message: 'Setting updated' };
        }
      }
    }
    
    sheet.appendRow([settingName, value, now, updatedBy || 'System']);
    return { success: true, message: 'Setting created' };
    
  } catch (e) {
    Logger.log("Error setting value: " + e.toString());
    return { success: false, error: e.toString() };
  }
}

function getAllSettings() {
  try {
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName(SETTINGS_SHEET);
    
    if (!sheet || sheet.getLastRow() < 2) {
      return {
        AUTO_ASSIGN: false,
        DRY_RUN: false,
        FORM_ENABLED: true,
        DASHBOARD_ENABLED: true,
        PRIMARY_ADMIN_EMAIL: ADMIN_EMAILS.PRIMARY,
        SECONDARY_ADMIN_EMAIL: ADMIN_EMAILS.SECONDARY,
        TERTIARY_ADMIN_EMAIL: ADMIN_EMAILS.TERTIARY,
        ESCALATION_EMAIL: ADMIN_EMAILS.ESCALATION,
        CRITICAL_EMAIL: ADMIN_EMAILS.CRITICAL
      };
    }
    
    var data = sheet.getRange(2, 1, sheet.getLastRow() - 1, 2).getValues();
    var settings = {};
    
    for (var i = 0; i < data.length; i++) {
      var name = data[i][0];
      var value = data[i][1];
      
      if (value === 'true') value = true;
      else if (value === 'false') value = false;
      
      settings[name] = value;
    }
    
    return settings;
  } catch (e) {
    Logger.log("Error getting all settings: " + e.toString());
    return {};
  }
}

// ==================== WEB APP ====================
function doGet(e) {
  var page = (e && e.parameter && e.parameter.page) ? e.parameter.page : 'form';
  
  var output;
  
  // First, verify spreadsheet access
  try {
    var ss = getSpreadsheet();
    Logger.log("Spreadsheet accessed successfully: " + ss.getName());
  } catch (ssError) {
    // Show configuration error page
    output = HtmlService.createHtmlOutput(getConfigErrorPageHtml(ssError.message));
    output.setTitle("Configuration Required");
    output.setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
    output.addMetaTag('viewport', 'width=device-width, initial-scale=1');
    return output;
  }
  
  switch (page) {
    case 'dashboard':
      if (!DASHBOARD_ENABLED) {
        output = HtmlService.createHtmlOutput(getDisabledPageHtml("Dashboard Unavailable", DASHBOARD_DISABLED_MESSAGE));
      } else {
        output = HtmlService.createHtmlOutputFromFile("dashboard");
      }
      output.setTitle("IT Support Control Center");
      break;
      
    case 'form':
    default:
      if (!FORM_ENABLED) {
        output = HtmlService.createHtmlOutput(getDisabledPageHtml("IT Support Unavailable", FORM_DISABLED_MESSAGE));
      } else {
        output = HtmlService.createHtmlOutputFromFile("form");
      }
      output.setTitle("IT Support Request");
      break;
  }
  
  output.setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
  output.addMetaTag('viewport', 'width=device-width, initial-scale=1');
  output.setSandboxMode(HtmlService.SandboxMode.IFRAME);
  
  return output;
}

// ==================== USER AUTHENTICATION ====================
function validateUser(employeeId, password) {
  try {
    Logger.log("Validating user: " + employeeId);
    
    if (!employeeId || !password) {
      return { success: false, message: 'Employee ID and Password are required' };
    }
    
    var ss = getSpreadsheet();
    var usersSheet = ss.getSheetByName(USERS_SHEET);
    
    if (!usersSheet) {
      Logger.log("Users sheet not found");
      return { success: false, message: 'System not configured. Please contact IT Admin.' };
    }
    
    var lastRow = usersSheet.getLastRow();
    if (lastRow < 2) {
      Logger.log("No users in Users sheet");
      return { success: false, message: 'No users configured. Please contact IT Admin.' };
    }
    
    var data = usersSheet.getRange(2, 1, lastRow - 1, 5).getValues();
    
    for (var i = 0; i < data.length; i++) {
      var rowEmployeeId = data[i][0] ? data[i][0].toString().trim() : '';
      var rowPassword = data[i][1] ? data[i][1].toString() : '';
      var displayName = data[i][2] ? data[i][2].toString() : 'User';
      var role = data[i][3] ? data[i][3].toString() : 'User';
      var status = data[i][4] ? data[i][4].toString() : 'Online';
      
      if (rowEmployeeId.toLowerCase() === employeeId.toLowerCase().trim()) {
        if (rowPassword === password) {
          if (status.toLowerCase() === 'inactive') {
            return { success: false, message: 'Your account is inactive. Please contact IT Admin.' };
          }
          
          usersSheet.getRange(i + 2, 5).setValue('Online');
          
          Logger.log("User authenticated: " + displayName);
          return {
            success: true,
            displayName: displayName,
            role: role,
            employeeId: rowEmployeeId,
            status: 'Online',
            message: 'Login successful'
          };
        } else {
          return { success: false, message: 'Invalid password' };
        }
      }
    }
    
    return { success: false, message: 'Employee ID not found' };
    
  } catch (error) {
    Logger.log("Error in validateUser: " + error.toString());
    return { success: false, message: 'Authentication error. Please try again.' };
  }
}

// ==================== UPDATE USER STATUS ====================
function updateUserStatus(employeeId, newStatus) {
  try {
    Logger.log("Updating status for " + employeeId + " to " + newStatus);
    
    if (!employeeId) {
      return { success: false, error: 'Employee ID is required' };
    }
    
    if (!newStatus) {
      return { success: false, error: 'New status is required' };
    }
    
    var normalizedStatus = newStatus;
    var statusLower = newStatus.toLowerCase().trim();
    
    if (statusLower === 'online' || statusLower === 'active') {
      normalizedStatus = 'Online';
    } else if (statusLower === 'offline' || statusLower === 'inactive') {
      normalizedStatus = 'Offline';
    } else if (statusLower === 'break' || statusLower === 'on break') {
      normalizedStatus = 'Break';
    } else if (statusLower === 'in meeting' || statusLower === 'meeting') {
      normalizedStatus = 'In Meeting';
    }
    
    if (VALID_USER_STATUSES.indexOf(normalizedStatus) === -1) {
      return { success: false, error: 'Invalid status. Must be one of: Online, Offline, Break, In Meeting' };
    }
    
    var ss = getSpreadsheet();
    var usersSheet = ss.getSheetByName(USERS_SHEET);
    
    if (!usersSheet) {
      return { success: false, error: 'Users sheet not found' };
    }
    
    var lastRow = usersSheet.getLastRow();
    if (lastRow < 2) {
      return { success: false, error: 'No users found' };
    }
    
    var data = usersSheet.getRange(2, 1, lastRow - 1, 5).getValues();
    
    for (var i = 0; i < data.length; i++) {
      var rowEmployeeId = data[i][0] ? data[i][0].toString().trim() : '';
      
      if (rowEmployeeId.toLowerCase() === employeeId.toLowerCase()) {
        usersSheet.getRange(i + 2, 5).setValue(normalizedStatus);
        updateITStaffStatus(data[i][2], normalizedStatus);
        
        Logger.log("Status updated for " + employeeId + " to " + normalizedStatus);
        clearDashboardCache();
        
        return {
          success: true,
          message: 'Status updated to ' + normalizedStatus,
          employeeId: employeeId,
          newStatus: normalizedStatus
        };
      }
    }
    
    return { success: false, error: 'User not found' };
    
  } catch (e) {
    Logger.log("Error in updateUserStatus: " + e.toString());
    return { success: false, error: e.toString() };
  }
}

// ==================== SAFARI-COMPATIBLE EMPLOYEE LOOKUP ====================
function lookupEmployee(email) {
  try {
    Logger.log("Looking up employee by email: " + email);
    
    var result = {
      found: false,
      empId: "",
      name: "",
      vipLevel: "Low"
    };
    
    if (!email || typeof email !== 'string' || email.trim() === '') {
      Logger.log("Invalid or empty email provided");
      return JSON.parse(JSON.stringify(result));
    }
    
    var ss = safeGetSpreadsheet();
    if (!ss) {
      Logger.log("Cannot access spreadsheet");
      return JSON.parse(JSON.stringify(result));
    }
    
    var directory = ss.getSheetByName(DIRECTORY_SHEET);
    if (!directory) {
      Logger.log("StaffDirectory sheet not found");
      return JSON.parse(JSON.stringify(result));
    }
    
    var lastRow = directory.getLastRow();
    if (lastRow < 2) {
      Logger.log("StaffDirectory has no data");
      return JSON.parse(JSON.stringify(result));
    }
    
    var allData = directory.getRange(2, 1, lastRow - 1, 4).getValues();
    var searchEmail = String(email).toLowerCase().trim();
    
    for (var i = 0; i < allData.length; i++) {
      var row = allData[i];
      var rowEmail = row[DIR_COL.EMAIL] ? String(row[DIR_COL.EMAIL]).toLowerCase().trim() : '';
      
      if (rowEmail === searchEmail) {
        var empId = row[DIR_COL.EMPLOYEE_ID] ? String(row[DIR_COL.EMPLOYEE_ID]).trim() : "";
        var fullName = row[DIR_COL.NAME] ? String(row[DIR_COL.NAME]).trim() : "";
        var vipLevel = row[DIR_COL.VIP_LEVEL] ? String(row[DIR_COL.VIP_LEVEL]).trim() : "Low";
        
        Logger.log("Employee found - ID: " + empId + ", Name: " + fullName + ", VIP: " + vipLevel);
        
        result = {
          found: true,
          empId: empId,
          name: fullName,
          vipLevel: vipLevel || "Low"
        };
        
        return JSON.parse(JSON.stringify(result));
      }
    }
    
    Logger.log("Employee not found for email: " + searchEmail);
    return JSON.parse(JSON.stringify(result));
    
  } catch (e) {
    Logger.log("Error in lookupEmployee: " + e.toString());
    return JSON.parse(JSON.stringify({
      found: false,
      empId: "",
      name: "",
      vipLevel: "Low"
    }));
  }
}

    return {

      empId: "",
      name: "",
      vipLevel: "Low",

    };
  }
}

// ==================== SUBMIT TICKET ====================
function submitTicket(formData) {
  try {
    Logger.log("=== SUBMIT TICKET ===");
    Logger.log("Form data: " + JSON.stringify(formData));
    
    if (!formData || typeof formData !== 'object') {
      throw new Error("Invalid form data");
    }
    
    var requiredFields = ['email', 'phone', 'location', 'shortDescription'];
    var missingFields = requiredFields.filter(function(field) {
      return !formData[field] || formData[field].toString().trim() === '';
    });
    
    if (missingFields.length > 0) {
      throw new Error("Missing required fields: " + missingFields.join(', '));
    }
    
    var emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(formData.email)) {
      throw new Error("Invalid email format");
    }
    
    var ss = getSpreadsheet();
    if (!ss) {
      throw new Error("Unable to access spreadsheet");
    }
    
    var sheet = ss.getSheetByName(FORM_SHEET);
    
    if (!sheet) {
      Logger.log("Creating Form_Responses sheet...");
      sheet = ss.insertSheet(FORM_SHEET);
      sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]);
      formatHeaderRow(sheet, HEADERS.length);
      setColumnWidths(sheet);
      sheet.setFrozenRows(1);
    }
    
    var employeeLookup = lookupEmployee(formData.email);
    Logger.log("Employee lookup result: " + JSON.stringify(employeeLookup));
    
    var finalName = '';
    var nameSource = '';
    
    if (employeeLookup.found === true && employeeLookup.name && employeeLookup.name.toString().trim() !== '') {
      finalName = employeeLookup.name.toString().trim();
      nameSource = 'StaffDirectory';
    } else if (formData.name && formData.name.toString().trim() !== '') {
      finalName = formData.name.toString().trim();
      nameSource = 'Form';
    } else {
      finalName = 'Anonymous';
      nameSource = 'Default';
    }
    
    var vipLevel = 'Low';
    if (employeeLookup.found === true && employeeLookup.vipLevel) {
      vipLevel = employeeLookup.vipLevel.toString().trim();
    }
    
    var issueType = (formData.issueType || '').toString().trim();
    var impactArea = (formData.workMode || formData.impactArea || '').toString().trim();
    
    var criticalFlag = formData.criticalFlag === true || formData.criticalFlag === 'true';
    
    var priority = 'Pending';
    
    var ticketId = generateTicketId(sheet);
    var timestamp = new Date();
    
    Logger.log("Creating ticket: " + ticketId + ", Name: " + finalName + ", Priority: " + priority + ", Critical: " + (criticalFlag ? 'Yes' : 'No'));
    
    var rowData = [
      timestamp,
      ticketId,
      employeeLookup.empId || "",
      finalName,
      formData.email.toString().trim(),
      formData.phone.toString().trim(),
      formData.location.toString().trim(),
      issueType,
      impactArea,
      (formData.shortDescription || 'No description').toString().trim(),
      (formData.additionalDescription || '').toString().trim(),
      "Open",
      priority,
      vipLevel,
      criticalFlag ? "true" : "false",
      "",
      "",
      "L1",
      "",
      "",
      timestamp,
      "",
      ""
    ];
    
    sheet.appendRow(rowData);
    var lastRow = sheet.getLastRow();
    formatDataRow(sheet, lastRow, rowData.length);
    
    var historyNotes = 'Ticket created - Awaiting priority assignment';
    if (criticalFlag) historyNotes += ' [Marked as Critical]';
    if (issueType) historyNotes += ' - Issue: ' + issueType;
    if (impactArea) historyNotes += ' - Impact: ' + impactArea;
    
    logHistory(ticketId, 'Created', '', 'Open', finalName, historyNotes);
    clearDashboardCache();
    
    sendTicketSubmittedEmail(ticketId, timestamp, finalName, formData, vipLevel, issueType, impactArea, criticalFlag);
    
    sendNewTicketNotificationToIT(ticketId, timestamp, employeeLookup, finalName, formData, priority, vipLevel, issueType, impactArea, criticalFlag);
    
    return {
      success: true,
      ticketId: ticketId,
      message: "Ticket created successfully",
      timestamp: timestamp.toISOString(),
      priority: priority,
      vipLevel: vipLevel,
      impactArea: impactArea,
      issueType: issueType,
      criticalFlag: criticalFlag,
      employeeName: finalName,
      nameSource: nameSource,
      employeeFound: employeeLookup.found
    };
    
  } catch (error) {
    Logger.log("Error in submitTicket: " + error.toString());
    return {
      success: false,
      error: error.message || "Failed to create ticket"
    };
  }
}

      success: true,
      message: 'Priority updated to ' + newPriority,
      ticketId: ticketId,
      oldPriority: oldPriority,
      newPriority: newPriority,
      updatedDate: now.toISOString()
    };

    Logger.log("Error in updateTicketPriority: " + error.toString());

// Add a note to a ticket - stored ONLY in Ticket_Notes sheet (not in Activity_Log)
function addTicketNote(ticketId, noteText, addedBy, noteType) {
  try {
    Logger.log("Adding note to ticket: " + ticketId + " by " + addedBy);
    
    if (!ticketId) {
      return { success: false, error: 'Ticket ID is required' };
    }
    
    if (!noteText || noteText.trim() === '') {
      return { success: false, error: 'Note text is required' };
    }
    
    if (!addedBy || addedBy.trim() === '') {
      return { success: false, error: 'Added by is required' };
    }
    
    // Validate note type
    var validNoteType = noteType && NOTE_TYPES.indexOf(noteType) !== -1 ? noteType : 'General';
    
    var ss = getSpreadsheet();
    
    // Verify ticket exists
    var formSheet = ss.getSheetByName(FORM_SHEET);
    if (!formSheet) {
      return { success: false, error: 'Form_Responses sheet not found' };
    }
    
    var ticketRow = findTicketRow(formSheet, ticketId);
    if (ticketRow === -1) {
      return { success: false, error: 'Ticket not found' };
    }
    
    // Get or create notes sheet
    var notesSheet = ss.getSheetByName(NOTES_SHEET);
    if (!notesSheet) {
      notesSheet = ss.insertSheet(NOTES_SHEET);
      notesSheet.getRange(1, 1, 1, NOTES_HEADERS.length).setValues([NOTES_HEADERS]);
      notesSheet.getRange(1, 1, 1, NOTES_HEADERS.length).setBackground("#1e40af").setFontColor("#ffffff").setFontWeight("bold");
      notesSheet.setFrozenRows(1);
      notesSheet.setColumnWidth(1, 180);
      notesSheet.setColumnWidth(2, 120);
      notesSheet.setColumnWidth(3, 400);
      notesSheet.setColumnWidth(4, 150);
      notesSheet.setColumnWidth(5, 120);
      Logger.log("Created Ticket_Notes sheet");
    }
    
    var now = new Date();
    
    // Add note to notes sheet ONLY (not to Activity_Log)
    var noteRow = [
      now,
      ticketId,
      noteText.trim(),
      addedBy.trim(),
      validNoteType
    ];
    
    notesSheet.appendRow(noteRow);
    
    var lastRow = notesSheet.getLastRow();
    notesSheet.getRange(lastRow, 1).setNumberFormat('MM/dd/yyyy hh:mm:ss AM/PM');
    
    // Also update the ticket's last updated time
    safeUpdateCell(formSheet, ticketRow, COLUMNS.LAST_UPDATED, now);
    formSheet.getRange(ticketRow, COLUMNS.LAST_UPDATED).setNumberFormat('MM/dd/yyyy hh:mm:ss AM/PM');
    
    // DO NOT log to Activity_Log - notes are kept separate
    
    clearDashboardCache();
    
    Logger.log("Note added successfully to ticket " + ticketId + " (stored in Ticket_Notes sheet only)");
    
    return {
      success: true,
      message: 'Note added successfully',
      ticketId: ticketId,
      noteType: validNoteType,
      addedBy: addedBy.trim(),
      timestamp: now.toISOString()
    };
    
  } catch (error) {
    Logger.log("Error in addTicketNote: " + error.toString());
    return { success: false, error: error.message || 'Failed to add note' };
  }
}

// Get all notes for a ticket from the separate Ticket_Notes sheet
function getTicketNotes(ticketId) {
  try {
    Logger.log("Getting notes for ticket: " + ticketId);
    
    if (!ticketId) {
      return [];
    }
    
    var ss = getSpreadsheet();
    var notesSheet = ss.getSheetByName(NOTES_SHEET);
    
    if (!notesSheet || notesSheet.getLastRow() < 2) {
      Logger.log("No notes found - sheet empty or doesn't exist");
      return [];
    }
    
    var data = notesSheet.getRange(2, 1, notesSheet.getLastRow() - 1, 5).getValues();
    var notes = [];
    
    for (var i = 0; i < data.length; i++) {
      var rowTicketId = data[i][1] ? data[i][1].toString().trim() : "";
      
      if (rowTicketId === ticketId.toString().trim()) {
        var timestamp = data[i][0];
        if (timestamp instanceof Date) {
          timestamp = timestamp.toISOString();
        } else if (timestamp) {
          try {
            timestamp = new Date(timestamp).toISOString();
          } catch (e) {
            timestamp = timestamp.toString();
          }
        }
        
        notes.push({
          timestamp: timestamp,
          ticketId: rowTicketId,
          note: data[i][2] ? data[i][2].toString() : "",
          addedBy: data[i][3] ? data[i][3].toString() : "",
          noteType: data[i][4] ? data[i][4].toString() : "General"
        });
      }
    }
    
    // Sort by timestamp descending (newest first)
    notes.sort(function(a, b) {
      try {
        return new Date(b.timestamp) - new Date(a.timestamp);
      } catch (e) {
        return 0;
      }
    });
    
    Logger.log("getTicketNotes: " + notes.length + " notes found for " + ticketId);
    return notes;
    
  } catch (e) {
    Logger.log("Error getting ticket notes: " + e.toString());
    return [];
  }
}

// ==================== SEARCH TICKETS ====================
function searchTickets(searchType, searchValue) {
  try {
    Logger.log("Searching tickets by " + searchType + ": " + searchValue);
    
    if (!searchValue || searchValue.trim() === '') {
      return { success: false, error: 'Search value is required', tickets: [] };
    }
    
    var allTickets = getAllTickets();
    
    if (!allTickets || allTickets.length === 0) {
      return { success: false, error: 'No tickets found', tickets: [] };
    }
    
    var searchLower = searchValue.toLowerCase().trim();
    var matchedTickets = [];
    
    switch (searchType) {
      case 'name':
        matchedTickets = allTickets.filter(function(ticket) {
          var name = (ticket['Name'] || '').toString().toLowerCase();
          return name.indexOf(searchLower) !== -1;
        });
        break;
        
      case 'email':
        matchedTickets = allTickets.filter(function(ticket) {
          var email = (ticket['Email Address'] || '').toString().toLowerCase();
          return email.indexOf(searchLower) !== -1;
        });
        break;
        
      case 'phone':
        var searchDigits = searchValue.replace(/\D/g, '');
        matchedTickets = allTickets.filter(function(ticket) {
          var phone = (ticket['Phone Number'] || '').toString().replace(/\D/g, '');
          return phone.indexOf(searchDigits) !== -1;
        });
        break;
        
      case 'location':
        matchedTickets = allTickets.filter(function(ticket) {
          var location = (ticket['Location'] || '').toString().toLowerCase();
          return location.indexOf(searchLower) !== -1;
        });
        break;
        
      case 'issueType':
        matchedTickets = allTickets.filter(function(ticket) {
          var issueType = (ticket['Issue Type'] || '').toString().toLowerCase();
          return issueType.indexOf(searchLower) !== -1;
        });
        break;
        
      case 'ticketId':
        matchedTickets = allTickets.filter(function(ticket) {
          var ticketId = (ticket['Ticket ID'] || '').toString().toLowerCase();
          return ticketId.indexOf(searchLower) !== -1;
        });
        break;
        
      default:
        return { success: false, error: 'Invalid search type', tickets: [] };
    }
    
    matchedTickets.sort(function(a, b) {
      try {
        return new Date(b['Created Date'] || 0) - new Date(a['Created Date'] || 0);
      } catch (e) {
        return 0;
      }
    });
    
    return {
      success: true,
      tickets: matchedTickets.slice(0, 50),
      count: matchedTickets.length
    };
    
  } catch (error) {
    Logger.log("Error in searchTickets: " + error.toString());
    return { success: false, error: error.message || 'Search failed', tickets: [] };
  }
}

function getAllTickets() {
  try {
    var cacheKey = 'dashboard_tickets';
    var cached = getCachedData(cacheKey);
    if (cached) {
      Logger.log("getAllTickets: Returning " + cached.length + " cached tickets");
      return cached;
    }
    
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName(FORM_SHEET);
    
    if (!sheet) {
      Logger.log("Form_Responses sheet not found");
      return [];
    }
    
    var lastRow = sheet.getLastRow();
    if (lastRow < 2) {
      Logger.log("No tickets in sheet");
      return [];
    }
    
    var lastCol = sheet.getLastColumn();
    var data = sheet.getRange(1, 1, lastRow, lastCol).getValues();
    var headers = data[0];
    var tickets = [];
    
    Logger.log("Sheet headers: " + JSON.stringify(headers));
    
    for (var i = 1; i < data.length; i++) {
      var row = data[i];
      var ticket = {};
      
      headers.forEach(function(header, j) {
        if (!header) return;
        
        var headerStr = header.toString().trim();
        var value = row[j];
        
        if (headerStr.toLowerCase().indexOf('date') !== -1 && value) {
          if (value instanceof Date) {
            value = value.toISOString();
          } else {
            try {
              var d = new Date(value);
              if (!isNaN(d.getTime())) {
                value = d.toISOString();
              } else {
                value = value.toString();
              }
            } catch (e) {
              value = value.toString();
            }
          }
        } else {
          value = value !== undefined && value !== null ? value.toString() : "";
        }
        
        ticket[headerStr] = value;
      });
      
      if (ticket["Ticket ID"] && ticket["Ticket ID"].trim() !== "") {
        tickets.push(ticket);
      }
    }
    
    tickets.sort(function(a, b) {
      try {
        var dateA = a["Created Date"] ? new Date(a["Created Date"]) : new Date(0);
        var dateB = b["Created Date"] ? new Date(b["Created Date"]) : new Date(0);
        return dateB - dateA;
      } catch (e) {
        return 0;
      }
    });
    
    setCachedData(cacheKey, tickets);
    
    Logger.log("getAllTickets: " + tickets.length + " tickets loaded from sheet");
    return tickets;
    
  } catch (e) {
    Logger.log("Error in getAllTickets: " + e.toString());
    return [];
  }
}

function getDashboardStats(tickets) {
  try {
    if (!tickets) tickets = getAllTickets();
    
    var stats = {
      total: tickets.length,
      open: 0,
      inProgress: 0,
      resolved: 0,
      unassigned: 0,
      highPriority: 0,
      critical: 0,
      pending: 0,
      markedCritical: 0,
      resolvedToday: 0,
      byLocation: {},
      byVipLevel: {},
      byAssignee: {},
      byEscalationLevel: { 'L1': 0, 'L2': 0, 'L3': 0 },
      byIssueType: {},
      byImpactArea: {}
    };
    
    var today = new Date().toDateString();
    
    tickets.forEach(function(ticket) {
      var status = (ticket['Status'] || 'Open').toString().trim();
      var location = (ticket['Location'] || 'Unknown').toString().trim();
      var vipLevel = (ticket['VIP Level'] || 'Low').toString().trim();
      var assignedTo = (ticket['Assigned To'] || 'Unassigned').toString().trim();
      var priority = (ticket['Priority'] || 'Pending').toString().trim();
      var escalationLevel = (ticket['Escalation Level'] || 'L1').toString().trim();
      var resolvedDate = ticket['Resolved Date'];
      var issueType = (ticket['Issue Type'] || '').toString().trim();
      var impactArea = (ticket['Impact Area'] || '').toString().trim();
      var criticalFlag = ticket['Critical Flag'] === 'true' || ticket['Critical Flag'] === true;
      
      switch (status.toLowerCase()) {
        case 'open': stats.open++; break;
        case 'in progress': stats.inProgress++; break;
        case 'resolved': stats.resolved++; break;
      }
      
      if (!ticket['Assigned To'] || ticket['Assigned To'] === '') {
        stats.unassigned++;
      }
      
      if (priority === 'High' || priority === 'Critical') {
        stats.highPriority++;
      }
      
      if (priority === 'Critical') {
        stats.critical++;
      }
      
      if (priority === 'Pending') {
        stats.pending++;
      }
      
      if (criticalFlag) {
        stats.markedCritical++;
      }
      
      if (resolvedDate) {
        try {
          var resDate = new Date(resolvedDate).toDateString();
          if (resDate === today) {
            stats.resolvedToday++;
          }
        } catch (e) {}
      }
      
      stats.byLocation[location] = (stats.byLocation[location] || 0) + 1;
      stats.byVipLevel[vipLevel] = (stats.byVipLevel[vipLevel] || 0) + 1;
      stats.byAssignee[assignedTo] = (stats.byAssignee[assignedTo] || 0) + 1;
      
      if (stats.byEscalationLevel.hasOwnProperty(escalationLevel)) {
        stats.byEscalationLevel[escalationLevel]++;
      }
      
      if (issueType) {
        stats.byIssueType[issueType] = (stats.byIssueType[issueType] || 0) + 1;
      }
      if (impactArea) {
        stats.byImpactArea[impactArea] = (stats.byImpactArea[impactArea] || 0) + 1;
      }
    });
    
    return stats;
    
  } catch (e) {
    Logger.log("Error in getDashboardStats: " + e.toString());
    return { total: 0, open: 0, inProgress: 0, resolved: 0, unassigned: 0 };
  }
}

    if (row === -1) {
      return { success: false, error: 'Ticket not found' };
    }
    
    var headers = sheet.getRange(1, 1, 1, TOTAL_COLUMNS).getValues()[0];
    var rowData = sheet.getRange(row, 1, 1, TOTAL_COLUMNS).getValues()[0];
    
    var ticket = {};
    headers.forEach(function(header, index) {
      var value = rowData[index];
      
      if (header && header.toString().toLowerCase().indexOf('date') !== -1 && value) {
        try {
          ticket[header] = value instanceof Date ? value.toISOString() : new Date(value).toISOString();
        } catch (e) {
          ticket[header] = value ? value.toString() : '';
        }
      } else {
        ticket[header] = value !== undefined && value !== null ? value : '';
      }
    });
    
    var history = getTicketHistory(ticketId);
    var notes = getTicketNotes(ticketId);
    
    return {
      success: true,
      ticket: ticket,
      history: history,
      notes: notes
    };
    
  } catch (error) {
    Logger.log("Error in getTicketByIdForTracking: " + error.toString());
    return { success: false, error: error.message || 'Failed to fetch ticket' };
  }
}

// ==================== UPDATE TICKET STATUS ====================
function updateTicketStatus(ticketId, newStatus, resolvedBy) {
  try {
    Logger.log("Updating status: " + ticketId + " to " + newStatus);
    
    if (!ticketId) {
      return { success: false, error: 'Ticket ID is required' };
    }
    
    if (!newStatus) {
      return { success: false, error: 'New status is required' };
    }
    
    var validStatuses = ['Open', 'In Progress', 'Resolved'];
    if (validStatuses.indexOf(newStatus) === -1) {
      return { success: false, error: 'Invalid status' };
    }
    
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName(FORM_SHEET);
    
    if (!sheet) {
      return { success: false, error: 'Form_Responses sheet not found' };
    }
    
    var row = findTicketRow(sheet, ticketId);
    
    if (row === -1) {
      return { success: false, error: 'Ticket not found' };
    }
    
    var now = new Date();
    var rowData = getRowData(sheet, row);
    var oldStatus = rowData[COLUMNS.STATUS - 1] || 'Open';
    var assignedTo = rowData[COLUMNS.ASSIGNED_TO - 1] || '';
    
    safeUpdateCell(sheet, row, COLUMNS.STATUS, newStatus);
    safeUpdateCell(sheet, row, COLUMNS.LAST_UPDATED, now);
    sheet.getRange(row, COLUMNS.LAST_UPDATED).setNumberFormat('MM/dd/yyyy hh:mm:ss AM/PM');
    
    if (newStatus === 'Resolved') {
      var resolver = resolvedBy || assignedTo || 'IT Support';
      safeUpdateCell(sheet, row, COLUMNS.RESOLVED_BY, resolver);
      safeUpdateCell(sheet, row, COLUMNS.RESOLVED_DATE, now);
      sheet.getRange(row, COLUMNS.RESOLVED_DATE).setNumberFormat('MM/dd/yyyy hh:mm:ss AM/PM');
      logHistory(ticketId, 'Resolved', oldStatus, 'Resolved', resolver, 'Ticket resolved by ' + resolver);
    } else if (oldStatus === 'Resolved') {
      safeUpdateCell(sheet, row, COLUMNS.RESOLVED_BY, '');
      safeUpdateCell(sheet, row, COLUMNS.RESOLVED_DATE, '');
      logHistory(ticketId, 'Reopened', 'Resolved', newStatus, resolvedBy || 'Dashboard User', 'Ticket reopened by ' + (resolvedBy || 'Dashboard User'));
    } else {
      logHistory(ticketId, 'Status Changed', oldStatus, newStatus, resolvedBy || 'Dashboard User', 'Status updated by ' + (resolvedBy || 'Dashboard User'));
    }
    
    clearDashboardCache();
    
    if (oldStatus !== newStatus) {
      var userEmail = rowData[COLUMNS.EMAIL - 1];
      var userName = rowData[COLUMNS.NAME - 1];
      
      if (userEmail) {
        var ticketInfo = {
          'Ticket ID': rowData[COLUMNS.TICKET_ID - 1],
          'Short Description': rowData[COLUMNS.SHORT_DESCRIPTION - 1],
          'Location': rowData[COLUMNS.LOCATION - 1],
          'Phone Number': rowData[COLUMNS.PHONE - 1],
          'Issue Type': rowData[COLUMNS.ISSUE_TYPE - 1],
          'Assigned To': assignedTo,
          'Resolved By': newStatus === 'Resolved' ? (resolvedBy || assignedTo) : '',
          'Status': newStatus
        };
        
        sendStatusChangeEmail(ticketId, userName, userEmail, oldStatus, newStatus, ticketInfo);
      }
    }
    
    return {
      success: true,
      message: 'Status updated to ' + newStatus,
      ticketId: ticketId,
      newStatus: newStatus,
      updatedDate: now.toISOString()
    };
    
  } catch (error) {
    Logger.log("Error in updateTicketStatus: " + error.toString());
    return { success: false, error: error.message || 'Failed to update status' };
  }
}

// ==================== ASSIGN TICKET ====================
function assignTicket(ticketId, assignedTo, assignedBy, priority) {
  try {
    Logger.log("Assigning ticket: " + ticketId + " to " + assignedTo + (priority ? " with priority " + priority : ""));
    
    if (!ticketId) {
      return { success: false, error: 'Ticket ID is required' };
    }
    
    if (!assignedTo) {
      return { success: false, error: 'Assignee is required' };
    }
    
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName(FORM_SHEET);
    
    if (!sheet) {
      return { success: false, error: 'Form_Responses sheet not found' };
    }
    
    var row = findTicketRow(sheet, ticketId);
    
    if (row === -1) {
      return { success: false, error: 'Ticket not found' };
    }
    
    var now = new Date();
    var rowData = getRowData(sheet, row);
    var previousAssignee = rowData[COLUMNS.ASSIGNED_TO - 1] || 'Unassigned';
    var currentStatus = rowData[COLUMNS.STATUS - 1];
    var oldPriority = rowData[COLUMNS.PRIORITY - 1] || 'Pending';
    
    safeUpdateCell(sheet, row, COLUMNS.ASSIGNED_TO, assignedTo);
    safeUpdateCell(sheet, row, COLUMNS.ASSIGNED_DATE, now);
    sheet.getRange(row, COLUMNS.ASSIGNED_DATE).setNumberFormat('MM/dd/yyyy hh:mm:ss AM/PM');
    
    if (currentStatus === 'Open' || !currentStatus) {
      safeUpdateCell(sheet, row, COLUMNS.STATUS, 'In Progress');
    }
    
    var prioritySet = false;
    if (priority && VALID_PRIORITIES.indexOf(priority) !== -1) {
      safeUpdateCell(sheet, row, COLUMNS.PRIORITY, priority);
      prioritySet = true;
      Logger.log("Priority set to: " + priority);
    }
    
    safeUpdateCell(sheet, row, COLUMNS.LAST_UPDATED, now);
    sheet.getRange(row, COLUMNS.LAST_UPDATED).setNumberFormat('MM/dd/yyyy hh:mm:ss AM/PM');
    
    var actionType = (!previousAssignee || previousAssignee === 'Unassigned' || previousAssignee === '') ? 'Assigned' : 'Reassigned';
    var historyNotes = 'Ticket ' + actionType.toLowerCase() + ' to ' + assignedTo + ' by ' + (assignedBy || 'Dashboard User');
    if (prioritySet) {
      historyNotes += ' with priority ' + priority;
    }
    
    logHistory(ticketId, actionType, previousAssignee, assignedTo, assignedBy || 'Dashboard User', historyNotes);
    
    if (prioritySet && oldPriority !== priority) {
      logHistory(ticketId, 'Priority Changed', oldPriority, priority, assignedBy || 'Dashboard User', 'Priority set to ' + priority + ' by ' + (assignedBy || 'Dashboard User'));
    }
    
    clearDashboardCache();
    
    var userEmail = rowData[COLUMNS.EMAIL - 1];
    var userName = rowData[COLUMNS.NAME - 1];
    
    if (userEmail) {
      sendTicketAssignedEmailToUser(ticketId, userName, assignedTo, rowData);
    }
    
    var assignedStaffEmail = getITStaffEmail(assignedTo);
    if (assignedStaffEmail) {
      if (prioritySet) {
        rowData[COLUMNS.PRIORITY - 1] = priority;
      }
      sendTicketAssignedEmailToStaff(ticketId, assignedTo, assignedBy || 'Dashboard', rowData);
    }
    
    return {
      success: true,
      message: 'Ticket assigned to ' + assignedTo + (prioritySet ? ' with priority ' + priority : ''),
      ticketId: ticketId,
      assignedTo: assignedTo,
      priority: prioritySet ? priority : oldPriority,
      assignedDate: now.toISOString()
    };
    
  } catch (error) {
    Logger.log("Error in assignTicket: " + error.toString());
    return { success: false, error: error.message || 'Failed to assign ticket' };
  }
}

// ==================== ESCALATE TICKET ====================
function escalateTicket(ticketId, escalateTo, escalationLevel, escalatedBy, reason) {
  try {
    Logger.log("Escalating ticket: " + ticketId + " to " + escalationLevel);
    
    if (!ticketId) {
      return { success: false, error: 'Ticket ID is required' };
    }
    
    if (!escalateTo) {
      return { success: false, error: 'Escalation target is required' };
    }
    
    if (!escalationLevel) {
      return { success: false, error: 'Escalation level is required' };
    }
    
    var validLevels = ['L1', 'L2', 'L3'];
    if (validLevels.indexOf(escalationLevel) === -1) {
      return { success: false, error: 'Invalid level' };
    }
    
    var ss = getSpreadsheet();
    var sheet = ss.getSheetByName(FORM_SHEET);
    
    if (!sheet) {
      return { success: false, error: 'Form_Responses sheet not found' };
    }
    
    var row = findTicketRow(sheet, ticketId);
    
    if (row === -1) {
      return { success: false, error: 'Ticket not found' };
    }
    
    var now = new Date();
    var rowData = getRowData(sheet, row);
    var previousLevel = rowData[COLUMNS.ESCALATION_LEVEL - 1] || 'L1';
    var previousAssignee = rowData[COLUMNS.ASSIGNED_TO - 1] || 'Unassigned';
    
    safeUpdateCell(sheet, row, COLUMNS.ESCALATION_LEVEL, escalationLevel);
    safeUpdateCell(sheet, row, COLUMNS.ESCALATED_TO, escalateTo);
    safeUpdateCell(sheet, row, COLUMNS.ESCALATION_DATE, now);
    sheet.getRange(row, COLUMNS.ESCALATION_DATE).setNumberFormat('MM/dd/yyyy hh:mm:ss AM/PM');
    safeUpdateCell(sheet, row, COLUMNS.ASSIGNED_TO, escalateTo);
    safeUpdateCell(sheet, row, COLUMNS.STATUS, 'In Progress');
    safeUpdateCell(sheet, row, COLUMNS.LAST_UPDATED, now);
    sheet.getRange(row, COLUMNS.LAST_UPDATED).setNumberFormat('MM/dd/yyyy hh:mm:ss AM/PM');
    
    var historyNotes = 'Escalated by ' + (escalatedBy || 'Dashboard User');
    if (reason) {
      historyNotes += ' - Reason: ' + reason;
    }
    
    logHistory(ticketId, 'Escalated',
      previousLevel + ' - ' + previousAssignee,
      escalationLevel + ' - ' + escalateTo,
      escalatedBy || 'Dashboard User',
      historyNotes);
    
    clearDashboardCache();
    
    var userEmail = rowData[COLUMNS.EMAIL - 1];
    var userName = rowData[COLUMNS.NAME - 1];
    
    if (userEmail) {
      sendTicketEscalatedEmailToUser(ticketId, userName, escalateTo, escalationLevel, reason, rowData);
    }
    
    var escalatedStaffEmail = getITStaffEmail(escalateTo);
    if (escalatedStaffEmail) {
      sendTicketEscalatedEmailToStaff(ticketId, escalateTo, escalationLevel, escalatedBy || 'Dashboard', reason, rowData);
    }
    
    var criticalFlag = rowData[COLUMNS.CRITICAL_FLAG - 1];
    var priority = rowData[COLUMNS.PRIORITY - 1];
    if (priority === 'Critical' || criticalFlag === 'true') {
      sendCriticalEscalationNotification(ticketId, escalateTo, escalationLevel, reason, rowData);
    }
    
    return {
      success: true,
      message: 'Ticket escalated to ' + escalationLevel + ' - ' + escalateTo,
      ticketId: ticketId,
      level: escalationLevel,
      escalatedTo: escalateTo,
      escalationDate: now.toISOString()
    };
    
  } catch (error) {
    Logger.log("Error in escalateTicket: " + error.toString());
    return { success: false, error: error.message || 'Failed to escalate ticket' };
  }
}

// ==================== AUTO-ASSIGNMENT ====================
function autoAssignTicket(ticketId, priority) {
  var autoAssignEnabled = getAutoAssignSetting();
  
  if (!autoAssignEnabled) {
    Logger.log("Auto-assign is disabled");
    return null;
  }
  
  try {
    Logger.log("Auto-assigning ticket " + ticketId);
    
    var itStaff = getITStaffList().filter(function(staff) {
      var isL1 = staff.level === 'L1';
      var isAvailable = ['Online', 'Active'].indexOf(staff.status) !== -1;
      return isL1 && isAvailable;
    });
    
    if (itStaff.length === 0) {
      Logger.log("No available L1 staff for auto-assignment");
      return null;
    }
    
    var workload = getStaffWorkload();
    
    var selectedStaff = null;
    var lowestLoad = Infinity;
    
    for (var i = 0; i < itStaff.length; i++) {
      var staff = itStaff[i];
      var load = workload[staff.name] || 0;
      
      if (load < lowestLoad) {
        lowestLoad = load;
        selectedStaff = staff;
      }
    }
    
    if (!selectedStaff) {
      return null;
    }
    
    var result = assignTicket(ticketId, selectedStaff.name, 'Auto-Assignment', null);
    
    if (result.success) {
      return selectedStaff.name;
    }
    
    return null;
    
  } catch (e) {
    Logger.log("Error in autoAssignTicket: " + e.toString());
    return null;
  }
}

function getStaffWorkload() {
  try {
    var tickets = getAllTickets();
    var workload = {};
    
    for (var i = 0; i < tickets.length; i++) {
      var ticket = tickets[i];
      var status = (ticket['Status'] || '').toString().toLowerCase();
      var assignedTo = ticket['Assigned To'] || '';
      
      if ((status === 'open' || status === 'in progress') && assignedTo) {
        workload[assignedTo] = (workload[assignedTo] || 0) + 1;
      }
    }
    
    return workload;
    
  } catch (e) {
    Logger.log("Error getting workload: " + e.toString());
    return {};
  }
}

// ==================== HISTORY LOGGING ====================
function logHistory(ticketId, action, fromValue, toValue, performedBy, notes) {
  try {
    var ss = getSpreadsheet();
    var historySheet = ss.getSheetByName(HISTORY_SHEET);
    
    if (!historySheet) {
      historySheet = ss.insertSheet(HISTORY_SHEET);
      historySheet.getRange(1, 1, 1, HISTORY_HEADERS.length).setValues([HISTORY_HEADERS]);
      historySheet.getRange(1, 1, 1, HISTORY_HEADERS.length).setFontWeight('bold').setBackground('#4CAF50').setFontColor('#fff');
      historySheet.setFrozenRows(1);
    }

      toValue || '',

      notes || ''
    ];
    
    historySheet.appendRow(rowData);
    
    var lastRow = historySheet.getLastRow();
    historySheet.getRange(lastRow, 1).setNumberFormat('MM/dd/yyyy hh:mm:ss AM/PM');

function getTicketHistory(ticketId) {
  try {
    Logger.log("getTicketHistory: " + ticketId);
    
    if (!ticketId) return [];
    
    var ss = getSpreadsheet();
    var historySheet = ss.getSheetByName(HISTORY_SHEET);
    
    if (!historySheet || historySheet.getLastRow() < 2) {
      return [];
    }
    
    var data = historySheet.getRange(2, 1, historySheet.getLastRow() - 1, 7).getValues();
    var history = [];
    
    for (var i = 0; i < data.length; i++) {
      var rowTicketId = data[i][1] ? data[i][1].toString().trim() : "";
      if (rowTicketId === ticketId.toString().trim()) {
        var timestamp = data[i][0];
        if (timestamp instanceof Date) {
          timestamp = timestamp.toISOString();
        } else if (timestamp) {
          try {
            timestamp = new Date(timestamp).toISOString();
          } catch (e) {
            timestamp = timestamp.toString();
          }
        }
        
        history.push({
          timestamp: timestamp,
          ticketId: rowTicketId,
          action: data[i][2] ? data[i][2].toString() : "",
          from: data[i][3] ? data[i][3].toString() : "",
          to: data[i][4] ? data[i][4].toString() : "",
          performedBy: data[i][5] ? data[i][5].toString() : "",
          notes: data[i][6] ? data[i][6].toString() : ""
        });
      }
    }
    
    history.sort(function(a, b) {
      try {
        return new Date(b.timestamp) - new Date(a.timestamp);
      } catch (e) {
        return 0;
      }
    });
    
    Logger.log("getTicketHistory: " + history.length + " entries found");
    return history;
    
  } catch (e) {
    Logger.log("Error getting ticket history: " + e.toString());
    return [];
  }
}

// ==================== GET TICKET TIMELINE (HISTORY ONLY - NO NOTES) ====================
function getTicketTimeline(ticketId) {
  try {
    Logger.log("Getting timeline for ticket: " + ticketId);
    
    if (!ticketId) {
      return [];
    }
    
    // Get ONLY history from Activity_Log - notes are kept separate
    var history = getTicketHistory(ticketId);
    
    // Mark all entries with type 'history'
    var timeline = history.map(function(h) {
      h.type = 'history';
      return h;
    });
    
    // Sort by timestamp descending (newest first)
    timeline.sort(function(a, b) {
      try {
        return new Date(b.timestamp) - new Date(a.timestamp);
      } catch (e) {
        return 0;
      }
    });
    
    Logger.log("getTicketTimeline: " + timeline.length + " entries found (notes excluded)");
    return timeline;
    
  } catch (e) {
    Logger.log("Error getting ticket timeline: " + e.toString());
    return [];
  }
}

function findTicketRow(sheet, ticketId) {
  if (!ticketId || !sheet) return -1;
  
  var lastRow = sheet.getLastRow();
  if (lastRow < 2) return -1;
  
  var ticketIds = sheet.getRange(2, COLUMNS.TICKET_ID, lastRow - 1, 1).getValues();
  
  for (var i = 0; i < ticketIds.length; i++) {
    if (ticketIds[i][0] && ticketIds[i][0].toString().trim() === ticketId.toString().trim()) {
      return i + 2;
    }
  }
  
  return -1;
}

  }
  
  try {
    sheet.getRange(row, column).setValue(value);
    return true;
  } catch (e) {
    Logger.log("Error updating cell: " + e.toString());
    return false;
  }
}

function generateTicketId(sheet) {
  try {
    var lastRow = sheet.getLastRow();
    if (lastRow < 2) return "SSPTKT-001";
    
    var data = sheet.getRange(2, COLUMNS.TICKET_ID, lastRow - 1, 1).getValues();
    var highestNum = 0;
    
    for (var i = 0; i < data.length; i++) {
      var ticketId = data[i][0];
      if (ticketId && typeof ticketId === 'string' && ticketId.indexOf('SSPTKT-') === 0) {
        var numPart = ticketId.split('-')[1];
        var num = parseInt(numPart);
        if (!isNaN(num) && num > highestNum) {
          highestNum = num;
        }
      }
    }
    
    var newNum = (highestNum + 1).toString();
    while (newNum.length < 3) {
      newNum = '0' + newNum;
    }
    
    return "SSPTKT-" + newNum;
    
  } catch (e) {
    Logger.log("Error generating ticket ID: " + e.toString());
    return "SSPTKT-" + new Date().getTime().toString().slice(-6);
  }
}

// ==================== EMAIL FUNCTIONS ====================
function sendEmail(to, subject, body, cc) {
  if (DRY_RUN) {
    Logger.log("DRY_RUN: Email skipped to " + to);
    Logger.log("Subject: " + subject);
    return true;
  }
  
  try {
    var toEmails = Array.isArray(to) ? to.filter(function(e) { return e && e.trim() !== ''; }) : [to].filter(function(e) { return e && e.trim() !== ''; });
    
    if (toEmails.length === 0) {
      Logger.log("No valid email recipients");
      return false;
    }
    
    var htmlBody = '<div style="font-family: ' + EMAIL_STYLE.fontFamily + '; font-size: ' + EMAIL_STYLE.fontSize + '; line-height: ' + EMAIL_STYLE.lineHeight + '; color: ' + EMAIL_STYLE.color + ';">' + body.replace(/\n/g, '<br>') + '</div>';
    
    var options = { htmlBody: htmlBody };
    if (cc && cc.length > 0) {
      var validCC = cc.filter(function(e) { return e && e.trim() !== ''; });
      if (validCC.length > 0) {
        options.cc = validCC.join(",");
      }
    }
    
    GmailApp.sendEmail(toEmails.join(","), subject, body, options);
    Logger.log("Email sent to: " + toEmails.join(", "));
    return true;
  } catch (e) {
    Logger.log("Error sending email: " + e.toString());
    return false;
  }
}

function sendNewTicketNotificationToIT(ticketId, timestamp, employeeLookup, userName, formData, priority, vipLevel, issueType, impactArea, criticalFlag) {
  var urgencyTag = '';
  if (criticalFlag) {
    urgencyTag = ' [MARKED CRITICAL]';
  } else if (impactArea === 'System Outage') {
    urgencyTag = ' [SYSTEM OUTAGE]';
  } else if (impactArea === 'Security / Access') {
    urgencyTag = ' [SECURITY]';
  }
  
  var subject = 'NEW IT Ticket [VIP:' + vipLevel + ']' + urgencyTag + ': ' + ticketId;
  
  var body = 'A new IT ticket has been submitted.\n\n';
  body += 'Ticket Information:\n';
  body += '--------------------------------------------------\n';
  body += 'Ticket ID: ' + ticketId + '\n';
  body += 'Priority: ' + priority + ' (Awaiting assignment)\n';
  body += 'VIP Level: ' + vipLevel + '\n';
  if (criticalFlag) body += 'User Marked as: CRITICAL\n';
  if (impactArea) body += 'Impact Area: ' + impactArea + '\n';
  body += 'Submitted: ' + formatDate(timestamp) + '\n\n';
  body += 'Employee Details:\n';
  body += 'Employee ID: ' + (employeeLookup.empId || 'Not in directory') + '\n';
  body += 'Name: ' + userName + '\n';
  body += 'Email: ' + formData.email + '\n';
  body += 'Phone: ' + formData.phone + '\n';
  body += 'Location: ' + formData.location + '\n';
  body += '\nIssue Details:\n';
  if (issueType) body += 'Issue Type: ' + issueType + '\n';
  body += 'Description:\n' + formData.shortDescription + '\n';
  if (formData.additionalDescription) {
    body += '\nAdditional Details:\n' + formData.additionalDescription + '\n';
  }
  body += '--------------------------------------------------\n\n';
  body += 'Please assign this ticket and set the appropriate priority.\n\n';
  body += 'IT Support System';
  
  var adminEmails = getAdminEmails();
  
  // Build recipients list from all configured admin emails
  var recipients = [];
  
  if (adminEmails.primary && adminEmails.primary.trim() !== '') {
    recipients.push(adminEmails.primary);
  }
  
  if (adminEmails.secondary && adminEmails.secondary.trim() !== '') {
    recipients.push(adminEmails.secondary);
  }
  
  if (adminEmails.tertiary && adminEmails.tertiary.trim() !== '') {
    recipients.push(adminEmails.tertiary);
  }
  
  // Add critical email if critical flag is set and not already in list
  if (criticalFlag && adminEmails.critical && adminEmails.critical.trim() !== '') {
    if (recipients.indexOf(adminEmails.critical) === -1) {
      recipients.push(adminEmails.critical);
    }
  }
  
  // Remove duplicates
  var uniqueRecipients = [];
  for (var i = 0; i < recipients.length; i++) {
    if (uniqueRecipients.indexOf(recipients[i]) === -1) {
      uniqueRecipients.push(recipients[i]);
    }
  }
  
  Logger.log("Sending new ticket notification to: " + uniqueRecipients.join(", "));
  sendEmail(uniqueRecipients, subject, body);
}

function generateReport(reportType) {
  try {
    Logger.log("Generating report: " + reportType);
    
    var tickets = getAllTickets();
    var now = new Date();
    
    var reportData;
    var headers;
    var reportTitle;
    
    switch (reportType) {
      case 'tickets':
        reportData = generateTicketsReport(tickets);
        headers = ['Ticket ID', 'Created Date', 'Name', 'Email', 'Phone', 'Location', 
                   'Issue Type', 'Impact Area', 'Description', 'Status', 'Priority', 'VIP Level', 
                   'Marked Critical', 'Assigned To', 'Escalation Level', 'Resolved By', 'Resolved Date'];
        reportTitle = 'All_Tickets_Report';
        break;
        
      case 'team':
        reportData = generateTeamReport();
        headers = ['Staff Name', 'Level', 'Status', 'Tickets Handled', 'Resolved', 'Open', 
                   'SLA Compliance %', 'Avg Resolution Time'];
        reportTitle = 'Team_Performance_Report';
        break;
        
      case 'sla':
        reportData = generateSLAReport(tickets);
        headers = ['Ticket ID', 'Created Date', 'Priority', 'Status', 'Assigned To', 
                   'Response Time (hrs)', 'Resolution Time (hrs)', 'SLA Status', 'SLA Threshold (hrs)'];
        reportTitle = 'SLA_Compliance_Report';
        break;
        
      case 'monthly':
        reportData = generateMonthlyReport(tickets);
        headers = ['Metric', 'Value', 'Details'];
        reportTitle = 'Monthly_Summary_Report';
        break;
        
      case 'notes':
        reportData = generateNotesReport();
        headers = ['Ticket ID', 'Timestamp', 'Note Type', 'Added By', 'Note'];
        reportTitle = 'Ticket_Notes_Report';
        break;
        
      default:
        return { success: false, error: 'Invalid report type' };
    }
    
    var csvContent = generateCSVContent(headers, reportData);
    var dateStr = Utilities.formatDate(now, Session.getScriptTimeZone(), "yyyyMMdd_HHmmss");
    var fileName = reportTitle + '_' + dateStr + '.csv';
    
    Logger.log("Report generated: " + fileName);
    
    return {
      success: true,
      fileName: fileName,
      csvContent: csvContent,
      recordCount: reportData.length,
      reportType: reportType,
      generatedAt: now.toISOString()
    };
    
  } catch (e) {
    Logger.log("Error generating report: " + e.toString());
    return { success: false, error: e.toString() };
  }
}

function generateCSVContent(headers, data) {
  try {
    var csv = headers.map(function(h) { return escapeCSVField(h); }).join(',') + '\n';
    
    for (var i = 0; i < data.length; i++) {
      csv += data[i].map(function(cell) { return escapeCSVField(cell); }).join(',') + '\n';
    }
    
    return csv;
  } catch (e) {
    Logger.log("Error generating CSV: " + e.toString());
    return '';
  }
}

function escapeCSVField(field) {
  if (field === null || field === undefined) return '';
  
  var str = field.toString();
  
  if (str.indexOf(',') !== -1 || str.indexOf('"') !== -1 || str.indexOf('\n') !== -1 || str.indexOf('\r') !== -1) {
    str = '"' + str.replace(/"/g, '""') + '"';
  }
  
  return str;
}

function generateSLAReport(tickets) {
  return tickets.map(function(t) {
    var createdDate = t['Created Date'] ? new Date(t['Created Date']) : null;
    var assignedDate = t['Assigned Date'] ? new Date(t['Assigned Date']) : null;
    var resolvedDate = t['Resolved Date'] ? new Date(t['Resolved Date']) : null;
    var priority = t['Priority'] || 'Pending';
    var slaThreshold = SLA_THRESHOLDS[priority] || SLA_THRESHOLDS['Pending'];
    
    var responseTime = '--';
    var resolutionTime = '--';
    var slaStatus = 'Pending';
    
    if (createdDate && assignedDate) {
      responseTime = ((assignedDate - createdDate) / (1000 * 60 * 60)).toFixed(2);
    }
    
    if (createdDate && resolvedDate) {
      var resHours = (resolvedDate - createdDate) / (1000 * 60 * 60);
      resolutionTime = resHours.toFixed(2);
      slaStatus = resHours <= slaThreshold.resolution ? 'Within SLA' : 'Breached';
    }
    
    return [
      t['Ticket ID'] || '',
      createdDate ? formatDate(createdDate) : '',
      priority,
      t['Status'] || '',
      t['Assigned To'] || 'Unassigned',
      responseTime,
      resolutionTime,
      slaStatus,
      slaThreshold.resolution
    ];
  });
}

function getTeamPerformance(filters) {
  try {
    var tickets = getAllTickets();
    var itStaff = getITStaffList();
    
    if (itStaff.length === 0) return [];
    
    var performance = {};
    itStaff.forEach(function(staff) {
      performance[staff.name] = {
        name: staff.name,
        level: staff.level || 'L1',
        status: staff.status || 'Online',
        ticketsHandled: 0,
        resolved: 0,
        openTickets: 0,
        withinSla: 0,
        breachedSla: 0,
        totalResolutionHours: 0
      };
    });
    
    tickets.forEach(function(ticket) {
      var assignedTo = (ticket['Assigned To'] || '').toString().trim();
      var status = (ticket['Status'] || 'Open').toString().trim();
      var priority = (ticket['Priority'] || 'Pending').toString().trim();
      var createdDate = ticket['Created Date'] ? new Date(ticket['Created Date']) : null;
      var resolvedDate = ticket['Resolved Date'] ? new Date(ticket['Resolved Date']) : null;
      
      if (assignedTo && performance[assignedTo]) {
        performance[assignedTo].ticketsHandled++;
        
        if (status.toLowerCase() === 'resolved') {
          performance[assignedTo].resolved++;
          
          if (resolvedDate && createdDate) {
            var resolutionHours = (resolvedDate - createdDate) / (1000 * 60 * 60);
            performance[assignedTo].totalResolutionHours += resolutionHours;
            
            var slaThreshold = SLA_THRESHOLDS[priority] || SLA_THRESHOLDS['Pending'];
            if (resolutionHours <= slaThreshold.resolution) {
              performance[assignedTo].withinSla++;
            } else {
              performance[assignedTo].breachedSla++;
            }
          }
        } else {
          performance[assignedTo].openTickets++;
        }
      }
    });
    
    var result = [];
    for (var key in performance) {
      if (performance.hasOwnProperty(key)) {
        var member = performance[key];
        var totalProcessed = member.withinSla + member.breachedSla;
        var slaCompliance = totalProcessed > 0 ? Math.round((member.withinSla / totalProcessed) * 100) : 100;
        
        var avgResolutionHours = member.resolved > 0 ? member.totalResolutionHours / member.resolved : 0;
        
        var avgResolutionDisplay = '--';
        if (avgResolutionHours > 0) {
          if (avgResolutionHours < 1) {
            avgResolutionDisplay = Math.round(avgResolutionHours * 60) + 'm';
          } else if (avgResolutionHours < 24) {
            avgResolutionDisplay = avgResolutionHours.toFixed(1) + 'h';
          } else {
            avgResolutionDisplay = (avgResolutionHours / 24).toFixed(1) + 'd';
          }
        }
        
        result.push({
          name: member.name,
          level: member.level,
          status: member.status,
          ticketsHandled: member.ticketsHandled,
          resolved: member.resolved,
          openTickets: member.openTickets,
          slaCompliance: slaCompliance,
          avgResolutionTime: avgResolutionDisplay
        });
      }
    }
    
    result.sort(function(a, b) { return b.ticketsHandled - a.ticketsHandled; });
    return result;
    
  } catch (error) {
    Logger.log("Error in getTeamPerformance: " + error.toString());
    return [];
  }
}

function getSlaSummary(filters) {
  try {
    var tickets = getAllTickets();
    var today = new Date();
    today.setHours(0, 0, 0, 0);
    
    var resolvedToday = 0;
    var breachedTickets = 0;
    var withinSlaTickets = 0;
    var totalResponseTimeHours = 0;
    var ticketsWithResponse = 0;
    
    tickets.forEach(function(ticket) {
      var priority = (ticket['Priority'] || 'Pending').toString().trim();
      var createdDate = ticket['Created Date'] ? new Date(ticket['Created Date']) : null;
      var assignedDate = ticket['Assigned Date'] ? new Date(ticket['Assigned Date']) : null;
      var resolvedDate = ticket['Resolved Date'] ? new Date(ticket['Resolved Date']) : null;
      
      if (resolvedDate) {
        var resDateOnly = new Date(resolvedDate);
        resDateOnly.setHours(0, 0, 0, 0);
        if (resDateOnly.getTime() === today.getTime()) {
          resolvedToday++;
        }
      }
      
      if (createdDate && assignedDate) {
        var responseHours = (assignedDate - createdDate) / (1000 * 60 * 60);
        totalResponseTimeHours += responseHours;
        ticketsWithResponse++;
      }
      
      if (resolvedDate && createdDate) {
        var resolutionHours = (resolvedDate - createdDate) / (1000 * 60 * 60);
        var slaThreshold = SLA_THRESHOLDS[priority] || SLA_THRESHOLDS['Pending'];
        if (resolutionHours <= slaThreshold.resolution) {
          withinSlaTickets++;
        } else {
          breachedTickets++;
        }
      }
    });
    
    var avgResponseHours = ticketsWithResponse > 0 ? Math.round(totalResponseTimeHours / ticketsWithResponse * 10) / 10 : 0;
    
    var totalProcessed = withinSlaTickets + breachedTickets;
    var slaCompliance = totalProcessed > 0 ? Math.round((withinSlaTickets / totalProcessed) * 100) : 100;
    
    var avgResponseDisplay = '--';
    if (avgResponseHours > 0) {
      if (avgResponseHours < 1) {
        avgResponseDisplay = Math.round(avgResponseHours * 60) + 'm';
      } else if (avgResponseHours < 24) {
        avgResponseDisplay = avgResponseHours.toFixed(1) + 'h';
      } else {
        avgResponseDisplay = (avgResponseHours / 24).toFixed(1) + 'd';
      }
    }
    
    return {
      slaCompliance: slaCompliance,
      avgResponseTime: avgResponseDisplay,
      breachedTickets: breachedTickets,
      resolvedToday: resolvedToday,
      totalTickets: tickets.length
    };
    
  } catch (error) {

// ==================== BULK UPDATE FUNCTIONS ====================
function bulkUpdateTickets(ticketIds, updateData, updatedBy) {
  try {
    Logger.log("Bulk updating " + ticketIds.length + " tickets");
    
    if (!ticketIds || ticketIds.length === 0) {
      return { success: false, error: 'No ticket IDs provided' };
    }
    
    if (!updateData) {
      return { success: false, error: 'No update data provided' };
    }
    
    var results = [];
    var successCount = 0;
    var failCount = 0;
    
    for (var i = 0; i < ticketIds.length; i++) {
      var ticketId = ticketIds[i];
      var result = { ticketId: ticketId, success: false };
      
      try {
        if (updateData.status) {
          var statusResult = updateTicketStatus(ticketId, updateData.status, updatedBy);
          result.statusUpdate = statusResult.success;
        }
        
        if (updateData.priority) {
          var priorityResult = updateTicketPriority(ticketId, updateData.priority, updatedBy);
          result.priorityUpdate = priorityResult.success;
        }
        
        if (updateData.assignedTo) {
          var assignResult = assignTicket(ticketId, updateData.assignedTo, updatedBy, updateData.priority);
          result.assignUpdate = assignResult.success;
        }
        
        if (updateData.issueType) {
          var issueResult = updateIssueType(ticketId, updateData.issueType, updatedBy);
          result.issueTypeUpdate = issueResult.success;
        }
        
        if (updateData.impactArea) {
          var impactResult = updateImpactArea(ticketId, updateData.impactArea, updatedBy);
          result.impactAreaUpdate = impactResult.success;
        }
        
        if (updateData.phone) {
          var phoneResult = updatePhoneNumber(ticketId, updateData.phone, updatedBy);
          result.phoneUpdate = phoneResult.success;
        }
        
        if (updateData.location) {
          var locationResult = updateLocation(ticketId, updateData.location, updatedBy);
          result.locationUpdate = locationResult.success;
        }
        
        result.success = true;
        successCount++;
        
      } catch (e) {
        result.error = e.toString();
        failCount++;
      }
      
      results.push(result);
    }
    
    clearDashboardCache();
    
    return {
      success: true,
      message: 'Bulk update completed: ' + successCount + ' succeeded, ' + failCount + ' failed',
      successCount: successCount,
      failCount: failCount,
      results: results
    };
    
  } catch (error) {
    Logger.log("Error in bulkUpdateTickets: " + error.toString());
    return { success: false, error: error.message || 'Bulk update failed' };
  }
}

// ==================== NOTIFICATION FUNCTIONS ====================
function getUnassignedTickets() {
  try {
    var tickets = getAllTickets();
    
    var unassigned = tickets.filter(function(ticket) {
      var status = ticket['Status'] || '';
      var assignedTo = ticket['Assigned To'] || '';
      
      return status !== 'Resolved' && (!assignedTo || assignedTo.trim() === '');
    });
    
    return {
      success: true,
      count: unassigned.length,
      tickets: unassigned
    };
    
  } catch (e) {
    Logger.log("Error getting unassigned tickets: " + e.toString());
    return { success: false, error: e.toString(), tickets: [] };
  }
}

function getOverdueTickets() {
  try {
    var tickets = getAllTickets();
    var now = new Date();
    var overdueTickets = [];
    
    tickets.forEach(function(ticket) {
      var status = ticket['Status'] || '';
      var priority = ticket['Priority'] || 'Pending';
      var createdDate = ticket['Created Date'] ? new Date(ticket['Created Date']) : null;
      
      if (status !== 'Resolved' && createdDate) {
        var slaThreshold = SLA_THRESHOLDS[priority] || SLA_THRESHOLDS['Pending'];
        var hoursSinceCreated = (now - createdDate) / (1000 * 60 * 60);
        
        if (hoursSinceCreated > slaThreshold.resolution) {
          ticket.hoursOverdue = Math.round(hoursSinceCreated - slaThreshold.resolution);
          overdueTickets.push(ticket);
        }
      }
    });
    
    // Sort by most overdue first
    overdueTickets.sort(function(a, b) {
      return b.hoursOverdue - a.hoursOverdue;
    });
    
    return {
      success: true,
      count: overdueTickets.length,
      tickets: overdueTickets
    };
    
  } catch (e) {
    Logger.log("Error getting overdue tickets: " + e.toString());
    return { success: false, error: e.toString(), tickets: [] };
  }
}

function dailyCleanup() {
  Logger.log("=== DAILY CLEANUP STARTED ===");
  
  try {
    // Clear old cache
    clearDashboardCache();
    
    // Log unassigned tickets
    var unassigned = getUnassignedTickets();
    Logger.log("Unassigned tickets: " + unassigned.count);
    
    // Log overdue tickets
    var overdue = getOverdueTickets();
    Logger.log("Overdue tickets: " + overdue.count);
    
    // Send daily summary if needed
    if (unassigned.count > 0 || overdue.count > 0) {
      sendDailySummaryEmail(unassigned, overdue);
    }
    
    Logger.log("=== DAILY CLEANUP COMPLETED ===");
    
    return {
      success: true,
      unassignedCount: unassigned.count,
      overdueCount: overdue.count
    };
    
  } catch (e) {
    Logger.log("Error in dailyCleanup: " + e.toString());
    return { success: false, error: e.toString() };
  }
}

function sendDailySummaryEmail(unassigned, overdue) {
  var adminEmails = getAdminEmails();
  if (!adminEmails.primary) return;
  
  var subject = 'IT Support Daily Summary - ' + formatDate(new Date());
  
  var body = 'Daily IT Support Summary\n';
  body += '========================\n\n';
  body += 'Unassigned Tickets: ' + unassigned.count + '\n';
  body += 'Overdue Tickets: ' + overdue.count + '\n\n';
  
  if (unassigned.count > 0) {
    body += 'Unassigned Ticket IDs:\n';
    unassigned.tickets.slice(0, 10).forEach(function(t) {
      body += '- ' + t['Ticket ID'] + ' (' + t['Priority'] + ')\n';
    });
    if (unassigned.count > 10) {
      body += '... and ' + (unassigned.count - 10) + ' more\n';
    }
    body += '\n';
  }
  
  if (overdue.count > 0) {
    body += 'Overdue Ticket IDs:\n';
    overdue.tickets.slice(0, 10).forEach(function(t) {
      body += '- ' + t['Ticket ID'] + ' (' + t.hoursOverdue + ' hours overdue)\n';
    });
    if (overdue.count > 10) {
      body += '... and ' + (overdue.count - 10) + ' more\n';
    }
    body += '\n';
  }
  
  body += '\nPlease review and take action.\n\nIT Support System';
  
  // Send to all admin emails
  var recipients = [];
  if (adminEmails.primary) recipients.push(adminEmails.primary);
  if (adminEmails.secondary) recipients.push(adminEmails.secondary);
  if (adminEmails.tertiary) recipients.push(adminEmails.tertiary);
  
  // Remove duplicates and empty strings
  var uniqueRecipients = [];
  for (var i = 0; i < recipients.length; i++) {
    if (recipients[i] && recipients[i].trim() !== '' && uniqueRecipients.indexOf(recipients[i]) === -1) {
      uniqueRecipients.push(recipients[i]);
    }
  }
  
  sendEmail(uniqueRecipients, subject, body);
}
